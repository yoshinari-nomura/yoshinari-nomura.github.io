mhc-misc-hdr-decode() の一部を

		    (if (not (re-search-forward "^X-SC-Subject:" nil t))
			()
		      (goto-char (point-min))
		      (if mew-decode-illegal-raw-subject
			  ;; delete x-mew: header
			  (if (re-search-forward "^X-Mew: " nil t)
			      (let (beg)
				(beginning-of-line)
				(setq beg (point))
				(forward-line 1)
				(re-search-forward "^[^ \t]" nil 'end)
				(delete-region beg (match-beginning 0))))
			;; decode raw JIS string
			(while (< (point) (point-max))
			  (if (looking-at "[^:]+:? *")
			      (goto-char (match-end 0)))
