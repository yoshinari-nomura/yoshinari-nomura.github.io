;; for Emacs 19.28
(or (fboundp 'facep)
    ;; Introduced in Emacs 19.29.
    (defun facep (x)
      "Return t if X is a face name or an internal face vector."
      (and (or (and (fboundp 'internal-facep) (internal-facep x))
              (and 
               (symbolp x) 
               (assq x (and (boundp 'global-face-data) global-face-data))))
          t)))
