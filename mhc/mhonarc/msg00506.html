.emacs に次の設定を加えることで回避できる問題ですので。

(define-process-argument-editing
  "/cvs\\.exe\\'"
  (lambda (x)
    (let ((command (car x))
	  (argument (cdr x)))
      (setq argument (cygnus-process-argument-quoting argument))
      (concat
       (unix-to-dos-filename command) " "
       (unix-to-dos-argument (mapconcat (function concat) argument " ")
			     nil nil nil nil)))))
